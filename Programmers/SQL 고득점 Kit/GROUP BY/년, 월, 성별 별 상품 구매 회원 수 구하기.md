```MySQL
SELECT YEAR(SALES_DATE) YEAR, MONTH(SALES_DATE) MONTH, GENDER, COUNT(DISTINCT u.USER_ID) AS 'USERS'
FROM USER_INFO u JOIN ONLINE_SALE s ON u.USER_ID = s.USER_ID AND GENDER IS NOT NULL
GROUP BY 1, 2, 3
ORDER BY 1, 2, 3
```
